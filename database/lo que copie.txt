import pyodbc
import streamlit as st

@st.cache_resource
def get_connection():
    # Esta es la llave para entrar a tu SQL Server local
    connection_string = (
        "DRIVER={ODBC Driver 17 for SQL Server};"
        "SERVER=localhost\\SQLEXPRESS;"
        "DATABASE=GymAdmin;"
        "Trusted_Connection=yes;"
    )
    return pyodbc.connect(connection_string)
